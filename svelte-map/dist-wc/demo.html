Intégration minimale 
<!-- 1. Charger les styles -->
<link rel="stylesheet"
      href="igo2-map.css">

<!-- 2. Placer le composant -->
<igo2-map
  style="display:block;
         width:100%;
         height:600px;">
</igo2-map>

<!-- 3. Charger le script (en bas) -->
<script src="igo2-map.iife.js"></script>
Intégration dans une grille 
<!-- Plein écran -->
<igo2-map
  style="width:100vw;
         height:100vh;"></igo2-map>

<!-- Dans un conteneur -->
<div style="height:500px">
  <igo2-map
    style="width:100%;
           height:100%;">
  </igo2-map>
</div>

<!-- Dans un CMS (WordPress, etc.) -->
<igo2-map
  class="wp-block-embed"
  style="height:450px;">
</igo2-map>
